function getPeriodRanges(cycleLogs){
const periodDays=cycleLogs.filter(l=>l.type==='period').map(l=>l.date).sort();
if(periodDays.length===0)return[];
const ranges=[];
let cur={start:periodDays[0],end:periodDays[0],flows:[]};
const first=cycleLogs.find(l=>l.date===periodDays[0]&&l.type==='period');
if(first&&first.flow)cur.flows.push(first.flow);
for(let i=1;i<periodDays.length;i++){
const prev=periodDays[i-1],curr=periodDays[i];
if(daysBetween(prev,curr)===1){
cur.end=curr;
const log=cycleLogs.find(l=>l.date===curr&&l.type==='period');
if(log&&log.flow)cur.flows.push(log.flow);
}else{
ranges.push(cur);
cur={start:curr,end:curr,flows:[]};
const log=cycleLogs.find(l=>l.date===curr&&l.type==='period');
if(log&&log.flow)cur.flows.push(log.flow);
}
}
ranges.push(cur);
return ranges;
}
function getCycleLengths(cycleLogs){const r=getPeriodRanges(cycleLogs);if(r.length<2)return[];const l=[];for(let i=0;i<r.length-1;i++)l.push({start:r[i].start,length:daysBetween(r[i].start,r[i+1].start)});return l}
function getPeriodLengths(cycleLogs){return getPeriodRanges(cycleLogs).map(r=>daysBetween(r.start,r.end)+1)}
function getCycleDayForDate(dateStr,cycleLogs){const r=getPeriodRanges(cycleLogs);if(r.length===0)return null;for(let i=r.length-1;i>=0;i--)if(r[i].start<=dateStr)return daysBetween(r[i].start,dateStr)+1;return null}
function getDaysSinceLastPeriod(cycleLogs){const r=getPeriodRanges(cycleLogs);if(r.length===0)return null;return daysBetween(r[r.length-1].start,getTodayDateString())}
